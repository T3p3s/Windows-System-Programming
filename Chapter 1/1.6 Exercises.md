#define SECURITY_WIN32
#include <Windows.h>
#include <stdio.h>
#include <security.h>
#pragma comment(lib, "Secur32.lib")


int main()
{
	wchar_t buffer[MAX_PATH] = L"";
	DWORD size = sizeof(buffer) / sizeof(wchar_t);

	BOOL getcompname = ::GetComputerName(buffer, &size);

	if (!getcompname)
	{
		printf("Error: %d\n", ::GetLastError());
	}
	printf("Computer name: %ls\n", buffer);

	wchar_t winDirBuffer[MAX_PATH] = L"";
	UINT winDirSize = sizeof(winDirBuffer) / sizeof(wchar_t);
	UINT getwindowsdir = ::GetWindowsDirectory(winDirBuffer, winDirSize);

	if (getwindowsdir == 0)
	{
		printf("Error: %d\n", ::GetLastError());
	}
	printf("Windows Directory: %ls\n", winDirBuffer);


	LARGE_INTEGER ticks;
	if (!QueryPerformanceCounter(&ticks))
	{
		printf("Error: %d\n", ::GetLastError());
	}
	printf("Frequency %lld\n", ticks);

	DWORD productType;
	BOOL productinfo = ::GetProductInfo(10, 0, 0, 0, &productType);
	if (!productinfo)
	{
		printf("Error: %d\n", ::GetLastError());
	}
	printf("Product type: %u\n", productType);


	wchar_t objNameBuffer[256] = L"";
	DWORD objNameSize = sizeof(objNameBuffer) / sizeof(wchar_t);
	BOOL computerobjname = ::GetComputerObjectNameW(NameDnsDomain, objNameBuffer, &objNameSize);
	if (!computerobjname)
	{
		printf("Error: %d\n", GetLastError());
	}
	printf("Computer Object Name: %ls\n", objNameBuffer);

	return 0;
}
